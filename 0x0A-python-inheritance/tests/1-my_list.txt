"""Unittest for print_sorted"""

import unittest
print_sorted = __import__('1-my_list').print_sorted

class TestSortedList(unittest.TestCase):
    """Run unittests for print_sorted"""

    def test_ordered_list(self):
        """Test an ordered list of integers."""
        ordered = [1, 2, 3, 4, 5]
        self.assertEqual(print_sorted(ordered), [1, 2, 3, 4, 5])

    def test_unordered_list(self):
        """Test an unordered list of integers."""
        unordered = [1, 4, 2, 3, 5]
        self.assertEqual(print_sorted(unordered), [1, 2, 3, 4, 5])

    def test_with_negatives(self):
        """Test a list with negative integers."""
        negative_int = [-4, 3, -2, 1, 5]
        self.assertEqual(print_sorted(negative_int), [-4, -2, 1, 3, 5])

    def test_empty_list(self):
        """Test an empty list."""
        empty = []
        self.assertEqual(print_sorted(empty), None)

    def test_one_element_list(self):
        """Test a list with a single element."""
        one_element = [5]
        self.assertEqual(print_sorted(one_element), [5])

    def test_all_negatives(self):
        """Test a string."""
        neg_list = [-1, -2, -3, -4, -5]
        self.assertEqual(print_sorted(neg_list), [-5, -4, -3, -2, -1])

if __name__ == '__main__':
    unittest.main()
